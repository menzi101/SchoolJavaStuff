        PreparedStatement ps;

        SQLStuff db = new SQLStuff();
        Connection con;
        
        try {
            
            con = (Connection) db.getConnection();
            
            String query = "UPDATE customer SET custSurname = ?, custNum = ?, custAddress = ? WHERE custName = ?";
            // Query for database
            ps = (PreparedStatement) con.prepareStatement(query);
            // Prepared statements
            String name = custNameField.getText();
            String surname = custSurnameField.getText();
            String telNum = custTelNumField.getText();
            String address = custAddressField.getText();
        
            ps.setString(1, surname);
            ps.setString(2, telNum);
            ps.setString(3, address);
            ps.setString(4, name);

    private void searchForRecordsActionPerformed(java.awt.event.ActionEvent evt) {                                                 
        PreparedStatement ps;
        ResultSet rs;
        
        SQLStuff db = new SQLStuff();
        Connection con;
        
        try {
            
            con = (Connection) db.getConnection();
            
            String query = "SELECT * FROM customer WHERE custName=? AND custSurname=?";
            ps = (PreparedStatement) con.prepareStatement(query);
            
            String name = custNameField.getText();
            String surname = custSurnameField.getText();

            ps.setString(1, name);
            ps.setString(2, surname);
           
            ps.execute();
            
            rs = ps.executeQuery();
        
            if (rs.next()) {
         
            custTelNumField.setText(rs.getString("custNum"));
            custAddressField.setText(rs.getString("custAddress"));  
            
            JOptionPane.showMessageDialog(null,"Update the required records and click on the update records button if necessary.", "Search for Record", 2);
            
            }else{
                JOptionPane.showMessageDialog(null,"Something went wrong", "Error of sorts", JOptionPane.ERROR_MESSAGE);
            }
            con.close();
        } catch (SQLException sqle) {
            System.out.println("Error " + sqle);
            
        } catch (ClassNotFoundException cnfe){
            System.out.println("Error " + cnfe);
        } catch (Exception ex) {
            Logger.getLogger(ENSCustomerProgram.class.getName()).log(Level.SEVERE, null, ex);
