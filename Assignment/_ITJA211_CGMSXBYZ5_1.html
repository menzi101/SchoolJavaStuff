    private void loginButtonActionPerformed(java.awt.event.ActionEvent evt) {                                            
        PreparedStatement ps;
        ResultSet rs;
        
        String uname = userNameField.getText();//Variable for username
        String pass = String.valueOf(passwordField.getPassword());//Variable for Password
        
        String query = "SELECT * FROM emeralduser WHERE name=? AND password=?";
        //Query that does the Database.
        try {
            ps = (PreparedStatement) MySQLConnection.getConnection().prepareStatement(query);
            
            ps.setString(1, uname);
            ps.setString(2, pass);
            
            rs = ps.executeQuery();
            
            if (rs.next()) {
                 JOptionPane.showMessageDialog(null,"Hi, "+uname+" You have successfully Logged In.", "Login Successful", 2);
            
            }else{
                JOptionPane.showMessageDialog(null,"Incorrect Username or Password", "Login Failed", 2);
            }
